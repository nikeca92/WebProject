ALTER TABLE VS_OWNER.VS_NGR ADD NGRID number;

ALTER TABLE VS_OWNER.VS_NGR ADD CONSTRAINT NGRID_UNQ UNIQUE (NGRID);

CREATE SEQUENCE VS_OWNER.VS_NGR_NGRID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

UPDATE VS_OWNER.VS_NGR SET NGRID = VS_OWNER.VS_NGR_NGRID_SEQ.nextval;

CREATE TRIGGER VS_OWNER.TRG_VS_NGR_INSERT
BEFORE INSERT ON VS_OWNER.VS_NGR
FOR EACH ROW
BEGIN
   SELECT VS_OWNER.VS_NGR_NGRID_SEQ.nextval INTO :new.NGRID FROM dual;
END;
/

COMMIT;
