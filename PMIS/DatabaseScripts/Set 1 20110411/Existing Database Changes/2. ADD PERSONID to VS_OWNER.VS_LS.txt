ALTER TABLE VS_OWNER.VS_LS ADD PersonID number;

ALTER TABLE VS_OWNER.VS_LS ADD CONSTRAINT PersonID_UNQ UNIQUE (PersonID);

CREATE SEQUENCE VS_OWNER.VS_LS_PersonID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

UPDATE VS_OWNER.VS_LS SET PersonID = VS_OWNER.VS_LS_PersonID_SEQ.nextval;

CREATE TRIGGER VS_OWNER.TRG_VS_LS_INSERT
BEFORE INSERT ON VS_OWNER.VS_LS
FOR EACH ROW
BEGIN
   SELECT VS_OWNER.VS_LS_PersonID_SEQ.nextval INTO :new.PersonID FROM dual;
END;
/

COMMIT;
