CREATE SEQUENCE VS_OWNER.VS_DOG_DOGID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

UPDATE VS_OWNER.VS_DOG SET DOGID = VS_OWNER.VS_DOG_DOGID_SEQ.nextval;

BEFORE INSERT ON VS_OWNER.VS_DOG
FOR EACH ROW
BEGIN
   SELECT VS_OWNER.VS_DOG_DOGID_SEQ.nextval INTO :new.DOGID FROM dual;
END;
/

COMMIT;
