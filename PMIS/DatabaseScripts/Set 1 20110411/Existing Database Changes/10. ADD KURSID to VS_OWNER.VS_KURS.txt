ALTER TABLE VS_OWNER.VS_KURS ADD KURSID number;

ALTER TABLE VS_OWNER.VS_KURS ADD CONSTRAINT KURSID_UNQ UNIQUE (KURSID);

CREATE SEQUENCE VS_OWNER.VS_KURS_KURSID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

UPDATE VS_OWNER.VS_KURS SET KURSID = VS_OWNER.VS_KURS_KURSID_SEQ.nextval;

CREATE TRIGGER VS_OWNER.TRG_VS_KURS_INSERT
BEFORE INSERT ON VS_OWNER.VS_KURS
FOR EACH ROW
BEGIN
   SELECT VS_OWNER.VS_KURS_KURSID_SEQ.nextval INTO :new.KURSID FROM dual;
END;
/

COMMIT;
/

ALTER TABLE VS_OWNER.VS_KURS MODIFY (KURSID number NOT NULL);
