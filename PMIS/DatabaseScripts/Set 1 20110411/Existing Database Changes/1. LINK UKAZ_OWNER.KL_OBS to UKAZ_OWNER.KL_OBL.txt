ALTER TABLE UKAZ_OWNER.KL_OBS ADD KOD_OBL number;

ALTER TABLE UKAZ_OWNER.KL_OBS ADD CONSTRAINT OBS_OBL_FK FOREIGN KEY (KOD_OBL) REFERENCES UKAZ_OWNER.KL_OBL (KOD_OBL);

DECLARE
   KOD_OBL number;
   
   cursor nma is
      SELECT KOD_NMA, KOD_OBL, KOD_OBS
      FROM UKAZ_OWNER.KL_NMA;
BEGIN
   FOR NMA_REC IN nma
   LOOP
      UPDATE UKAZ_OWNER.KL_OBS SET
         KOD_OBL = NMA_REC.KOD_OBL
      WHERE UKAZ_OWNER.KL_OBS.KOD_OBS = NMA_REC.KOD_OBS;
   END LOOP;
END;
/

COMMIT;
