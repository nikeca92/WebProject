
CREATE TABLE PMIS_APPL.Exams (ExamID number NOT NULL,
                              ExamName varchar2(300)
                              );

ALTER TABLE PMIS_APPL.Exams ADD CONSTRAINT ExamID_PK PRIMARY KEY(ExamID);

CREATE SEQUENCE PMIS_APPL.Exams_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_APPL.Exams_INSERT
BEFORE INSERT ON PMIS_APPL.Exams
FOR EACH ROW
BEGIN
   SELECT PMIS_APPL.Exams_ID_SEQ.nextval INTO :new.ExamID FROM dual;
END;
/
