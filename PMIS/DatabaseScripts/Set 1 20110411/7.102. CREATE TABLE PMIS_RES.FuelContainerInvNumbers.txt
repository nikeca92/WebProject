CREATE TABLE PMIS_RES.FuelContainerInvNumbers(FuelContainerInvNumberID number NOT NULL,
                                        FuelContainerID number NOT NULL,
                                        InventoryNumber varchar2(20),
                                        ChangeDate date
                                        );

ALTER TABLE PMIS_RES.FuelContainerInvNumbers ADD CONSTRAINT FuelContainerInvNumbers_PK PRIMARY KEY(FuelContainerInvNumberID);
ALTER TABLE PMIS_RES.FuelContainerInvNumbers ADD CONSTRAINT FuelConInvNumb_FuelCont_FK FOREIGN KEY (FuelContainerID) REFERENCES PMIS_RES.FuelContainers (FuelContainerID);

CREATE SEQUENCE PMIS_RES.FuelContainerInvNumbers_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_RES.TRG_FuelContInvNum_INSERT
BEFORE INSERT ON PMIS_RES.FuelContainerInvNumbers
FOR EACH ROW
BEGIN
   SELECT PMIS_RES.FuelContainerInvNumbers_ID_SEQ.nextval INTO :new.FuelContainerInvNumberID FROM dual;
END;
