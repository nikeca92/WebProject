CREATE TABLE PMIS_ADM.ChangeEventTypes(ChangeEventTypeID number NOT NULL,
                                       ChangeEventTypeKey varchar2(500),
                                       ChangeEventType varchar2(300),
                                       ModuleID decimal NOT NULL,
                                      ChangeType decimal(2) NOT NULL CHECK (ChangeType BETWEEN 1 AND 3) /*1 = Edit; 2 = Insert; 3 = Delete;*/
                                      );


ALTER TABLE PMIS_ADM.ChangeEventTypes ADD CONSTRAINT ChangeEventTypes_PK PRIMARY KEY(ChangeEventTypeID);
ALTER TABLE PMIS_ADM.ChangeEventTypes ADD CONSTRAINT ChangeEventTypes_Modules_FK FOREIGN KEY (ModuleID) REFERENCES PMIS_ADM.Modules (ModuleID);
ALTER TABLE PMIS_ADM.ChangeEventTypes ADD CONSTRAINT ChangeEventTypes_UNQ UNIQUE (ChangeEventTypeKey);



CREATE SEQUENCE PMIS_ADM.ChangeEventTypes_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_ADM.TRG_ChangeEventTypes_INSERT
BEFORE INSERT ON PMIS_ADM.ChangeEventTypes
FOR EACH ROW
BEGIN
   SELECT PMIS_ADM.ChangeEventTypes_ID_SEQ.nextval INTO :new.ChangeEventTypeID FROM dual;
END;
