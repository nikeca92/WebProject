CREATE TABLE PMIS_ADM.OwnershipTypes(OwnershipTypeID number NOT NULL,
                                     OwnershipTypeKey varchar2(300),
                                     OwnershipTypeName varchar2(300)
                                    );

ALTER TABLE PMIS_ADM.OwnershipTypes ADD CONSTRAINT OwnershipTypes_PK PRIMARY KEY(OwnershipTypeID);

CREATE SEQUENCE PMIS_ADM.OwnershipTypes_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_ADM.TRG_OwnershipTypes_INSERT
BEFORE INSERT ON PMIS_ADM.OwnershipTypes
FOR EACH ROW
BEGIN
   SELECT PMIS_ADM.OwnershipTypes_ID_SEQ.nextval INTO :new.OwnershipTypeID FROM dual;
END;
/
