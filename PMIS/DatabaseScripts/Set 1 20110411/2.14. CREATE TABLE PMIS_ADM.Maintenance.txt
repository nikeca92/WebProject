ALTER TABLE PMIS_ADM.Maintenance ADD CONSTRAINT Maintenance_UNQ UNIQUE (MaintKey);

CREATE SEQUENCE PMIS_ADM.Maintenance_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_ADM.TRG_Maintenance_INSERT
BEFORE INSERT ON PMIS_ADM.Maintenance
FOR EACH ROW
BEGIN
   SELECT PMIS_ADM.Maintenance_ID_SEQ.nextval INTO :new.MaintID FROM dual;
END;
/
