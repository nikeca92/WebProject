CREATE TABLE PMIS_APPL.Specializations(SpecializationID number NOT NULL,
									   SpecializationName varchar2(500 byte) NOT NULL,
									   MilitarySchoolSubjectID number NOT NULL
									  );

ALTER TABLE PMIS_APPL.Specializations ADD CONSTRAINT Specializations_PK PRIMARY KEY(SpecializationID);
ALTER TABLE PMIS_APPL.Specializations ADD CONSTRAINT Specializations_KL_VSP_FK FOREIGN KEY (MilitarySchoolSubjectID) REFERENCES VS_OWNER.KLV_VSP (VSPID);

CREATE SEQUENCE PMIS_APPL.Specializations_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_APPL.TRG_Specializations_INSERT
BEFORE INSERT ON PMIS_APPL.Specializations
FOR EACH ROW
BEGIN
   SELECT PMIS_APPL.Specializations_ID_SEQ.nextval INTO :new.SpecializationID FROM dual;
END;
/
