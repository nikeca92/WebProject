CREATE TABLE PMIS_APPL.ApplicantExamStatuses(ApplicantExamStatusID number NOT NULL,
                                             ApplicantExamStatusName varchar2(500),
                                             ApplicantExamStatusKey varchar2(500)
                                            );

ALTER TABLE PMIS_APPL.ApplicantExamStatuses ADD CONSTRAINT ApplicantExamStatuses_PK PRIMARY KEY(ApplicantExamStatusID);

CREATE SEQUENCE PMIS_APPL.AppExamStatuses_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_APPL.TRG_AppExamStatuses_INSERT
BEFORE INSERT ON PMIS_APPL.ApplicantExamStatuses
FOR EACH ROW
BEGIN
   SELECT PMIS_APPL.AppExamStatuses_ID_SEQ.nextval INTO :new.ApplicantExamStatusID FROM dual;
END;
/


INSERT INTO PMIS_APPL.ApplicantExamStatuses(ApplicantExamStatusName, ApplicantExamStatusKey)
VALUES ('Класиран', 'RATED');

INSERT INTO PMIS_APPL.ApplicantExamStatuses(ApplicantExamStatusName, ApplicantExamStatusKey)
VALUES ('Некласиран', 'NOTRATED');

COMMIT;
