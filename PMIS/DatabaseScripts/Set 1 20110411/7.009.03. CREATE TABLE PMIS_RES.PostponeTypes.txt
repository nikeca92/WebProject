CREATE TABLE PMIS_RES.PostponeTypes(PostponeTypeID number NOT NULL,
                                    PostponeTypeKey varchar2(100),
                                    PostponeTypeName varchar2(250)
                                   );

ALTER TABLE PMIS_RES.PostponeTypes ADD CONSTRAINT PostponeTypes_PK PRIMARY KEY(PostponeTypeID);

CREATE SEQUENCE PMIS_RES.PostponeTypes_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_RES.TRG_PostponeTypes_INSERT
BEFORE INSERT ON PMIS_RES.PostponeTypes
FOR EACH ROW
BEGIN
   SELECT PMIS_RES.PostponeTypes_ID_SEQ.nextval INTO :new.PostponeTypeID FROM dual;
END;
/

INSERT INTO PMIS_RES.PostponeTypes (PostponeTypeKey, PostponeTypeName)
VALUES ('CONDITIONED', 'Условно');

INSERT INTO PMIS_RES.PostponeTypes (PostponeTypeKey, PostponeTypeName)
VALUES ('ABSOLUTELY', 'Безусловно');



COMMIT;
/
