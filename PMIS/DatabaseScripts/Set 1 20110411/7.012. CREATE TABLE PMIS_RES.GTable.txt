ALTER TABLE PMIS_RES.GTable ADD CONSTRAINT APPL_GTable_UNQ UNIQUE (TableName, TableKey);

CREATE SEQUENCE PMIS_RES.GTable_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_RES.TRG_GTable_INSERT
BEFORE INSERT ON PMIS_RES.GTable
FOR EACH ROW
DECLARE
   TableKey number;
BEGIN
   SELECT PMIS_RES.GTable_ID_SEQ.nextval INTO :new.GTableID FROM dual;
   
   IF :new.TableKey IS NULL THEN
      SELECT MAX(TableKey) INTO TableKey
      FROM PMIS_RES.GTable
      WHERE TableName = :new.TableName;
      
      :new.TableKey := TableKey + 1;
   END IF;
END;
