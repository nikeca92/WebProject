                                    );

ALTER TABLE PMIS_RES.TechnicsTypes ADD CONSTRAINT TechnicsTypes_UNQ UNIQUE (TechnicsTypeKey);

CREATE SEQUENCE PMIS_RES.TechnicsTypes_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_RES.TRG_TechnicsTypes_INSERT
BEFORE INSERT ON PMIS_RES.TechnicsTypes
FOR EACH ROW
BEGIN
   SELECT PMIS_RES.TechnicsTypes_ID_SEQ.nextval INTO :new.TechnicsTypeID FROM dual;
END;
