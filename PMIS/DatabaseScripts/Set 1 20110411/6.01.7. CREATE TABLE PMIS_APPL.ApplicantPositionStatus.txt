
CREATE TABLE PMIS_APPL.ApplicantPositionStatus (StatusID number NOT NULL,
                                                StatusName varchar2(300),
			                        StatusKey varchar2(300)
                                               );

ALTER TABLE PMIS_APPL.ApplicantPositionStatus ADD CONSTRAINT AppPosStatusID_PK PRIMARY KEY(StatusID);

CREATE SEQUENCE PMIS_APPL.ApplicantPosStatusID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_APPL.ApplicantPosStatus_INSERT
BEFORE INSERT ON PMIS_APPL.ApplicantPositionStatus
FOR EACH ROW
BEGIN
   SELECT PMIS_APPL.ApplicantPosStatusID_SEQ.nextval INTO :new.StatusID FROM dual;
END;
/
