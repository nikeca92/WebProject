CREATE SEQUENCE PMIS_ADM.CHANGETYPES_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

BEFORE INSERT ON PMIS_ADM.ChangeTypes
FOR EACH ROW
BEGIN

END;
