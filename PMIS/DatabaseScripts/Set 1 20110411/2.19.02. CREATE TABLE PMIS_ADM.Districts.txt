CREATE TABLE PMIS_ADM.Districts(DistrictID number NOT NULL,
                                DistrictName varchar2(300),
                                PostCode varchar2(10),
                                CityID number NOT NULL
                               );

ALTER TABLE PMIS_ADM.Districts ADD CONSTRAINT Districts_PK PRIMARY KEY(DistrictID);
ALTER TABLE PMIS_ADM.Districts ADD CONSTRAINT Districts_Cities_FK FOREIGN KEY (CityID) REFERENCES UKAZ_OWNER.KL_NMA (KOD_NMA);

CREATE SEQUENCE PMIS_ADM.Districts_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_ADM.TRG_Districts_INSERT
BEFORE INSERT ON PMIS_ADM.Districts
FOR EACH ROW
BEGIN
   SELECT PMIS_ADM.Districts_ID_SEQ.nextval INTO :new.DistrictID FROM dual;
END;
