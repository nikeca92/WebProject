CREATE TABLE PMIS_HS.ProtocolTypes(ProtocolTypeID number NOT NULL,
                                   ProtocolTypeName varchar2(500)
                                  );

ALTER TABLE PMIS_HS.ProtocolTypes ADD CONSTRAINT ProtocolTypes_PK PRIMARY KEY(ProtocolTypeID);

CREATE SEQUENCE PMIS_HS.ProtocolTypes_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_HS.TRG_ProtocolTypes_INSERT
BEFORE INSERT ON PMIS_HS.ProtocolTypes
FOR EACH ROW
BEGIN
   SELECT PMIS_HS.ProtocolTypes_ID_SEQ.nextval INTO :new.ProtocolTypeID FROM dual;
END;
/
