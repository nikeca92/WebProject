CREATE TABLE PMIS_HS.Probabilities(ProbabilityID number NOT NULL,
                                   ProbabilityName varchar2(500),
                                   ProbabilityFactor decimal(30, 10));

ALTER TABLE PMIS_HS.Probabilities ADD CONSTRAINT Probabilities_PK PRIMARY KEY(ProbabilityID);

CREATE SEQUENCE PMIS_HS.Probabilities_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_HS.TRG_Probabilities_INSERT
BEFORE INSERT ON PMIS_HS.Probabilities
FOR EACH ROW
BEGIN
   SELECT PMIS_HS.Probabilities_ID_SEQ.nextval INTO :new.ProbabilityID FROM dual;
END;
/
