CREATE TABLE PMIS_HS.Exposure(ExposureID number NOT NULL,
                              ExposureName varchar2(500),
                              ExposureFactor decimal(30, 10));

ALTER TABLE PMIS_HS.Exposure ADD CONSTRAINT Exposure_PK PRIMARY KEY(ExposureID);

CREATE SEQUENCE PMIS_HS.Exposure_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_HS.TRG_Exposure_INSERT
BEFORE INSERT ON PMIS_HS.Exposure
FOR EACH ROW
BEGIN
   SELECT PMIS_HS.Exposure_ID_SEQ.nextval INTO :new.ExposureID FROM dual;
END;
