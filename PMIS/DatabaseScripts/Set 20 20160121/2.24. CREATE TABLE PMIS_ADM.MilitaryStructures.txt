CREATE TABLE PMIS_ADM.MilitaryStructures(MilitaryStructureID number NOT NULL,
                                         MilitaryStructureName varchar2(500),
                                         IsActive number(1) CHECK (IsActive BETWEEN 0 AND 1),
                                         Seq number
                                        );

ALTER TABLE PMIS_ADM.MilitaryStructures ADD CONSTRAINT MilitaryStructures_PK PRIMARY KEY(MilitaryStructureID);

CREATE SEQUENCE PMIS_ADM.MilitaryStructures_ID_SEQ
START WITH 1 
INCREMENT BY 1 
NOMAXVALUE;

CREATE TRIGGER PMIS_ADM.TRG_MilitaryStructures_INSERT
BEFORE INSERT ON PMIS_ADM.MilitaryStructures
FOR EACH ROW
BEGIN
   SELECT PMIS_ADM.MilitaryStructures_ID_SEQ.nextval INTO :new.MilitaryStructureID FROM dual;
END;
/
